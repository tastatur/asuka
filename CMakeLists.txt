cmake_minimum_required(VERSION 3.0.0)
project(euterpe VERSION 0.1.0)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

find_package(Boost REQUIRED system)
include_directories(${Boost_INCLUDE_DIR})

include_directories(/usr/local/include/essentia/)
include_directories(/usr/local/include/essentia/scheduler/)
include_directories(/usr/local/include/essentia/streaming/)

include_directories(/usr/local/include/gaia2)
include_directories(/usr/include/Qt)
include_directories(/usr/include/QtCore)
include_directories(/usr/include/eigen3)

add_executable(train train.cpp)

link_directories(/usr/local/lib)
target_link_libraries(train essentia)
target_link_libraries(train fftw3)
target_link_libraries(train yaml)
target_link_libraries(train avcodec)
target_link_libraries(train avformat)
target_link_libraries(train avutil)
target_link_libraries(train samplerate)

target_link_libraries(train fftw3f)

target_link_libraries(train ${Boost_LIBRARIES})
